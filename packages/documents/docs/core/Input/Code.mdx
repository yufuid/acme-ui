## 位置聚焦

聚焦额外配置属性

<Playground>
  {() => {
    const inputRef = React.createRef();
    const jumpToStart = () => {
      inputRef.selectionStart = 0;
      inputRef.current.focus();
      inputRef.current.setSelectionRange(0, 0);
    };
    const jumpToEnd = () => {
      const valueLen = inputRef.current.value.length;
      inputRef.current.focus();
      inputRef.current.setSelectionRange(valueLen, valueLen);
    };
    const selectAll = () => {
      inputRef.current.select();
    };
    return (
      <div>
        <Input
          ref={inputRef}
          placeholder="请输入内容"
          style={{ width: '400px' }}
          value="这是一个输入框"
        />
        <div
          style={{
            width: '400px',
            marginTop: '24px',
            display: 'flex',
            justifyContent: 'space-between',
          }}
        >
          <Button mode="outlined" size="small" onClick={jumpToStart}>
            开始位置
          </Button>
          <Button mode="outlined" size="small" onClick={selectAll}>
            聚焦并全选
          </Button>
          <Button mode="outlined" size="small" onClick={jumpToEnd}>
            结束位置
          </Button>
        </div>
      </div>
    );
  }}
</Playground>
