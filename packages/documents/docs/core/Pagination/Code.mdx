import { Playground, Props } from "~controls";
import { Pagination } from "@acme-ui/core";

## 演示
<Playground
    links={{
        codeSandbox: 'https://www.baidu.com',
        codePen: 'https://www.baidu.com',
        stackblitz: 'https://www.baidu.com',
    }}
    defaultComProps={{
        type: 'default',
        showTotal: false,
        showJump: false,
        showPageSize: false,
        total: 50,
    }}
    headWrapper={Playground.renderChecker([
        {label: '基本', onClick: (p) => ({...p, type: 'default', total: 50}), highlight: (p) => p.type === 'default' && p.total === 50},
        {label: '更多', onClick: (p) => ({...p, type: 'default', total: 200}), highlight: (p) => p.type === 'default' && p.total === 200},
        {label: '迷你版本', onClick: (p) => ({...p, type: 'mini', total: 50}), highlight: (p) => p.type === 'mini'},
        {label: '简洁版本', onClick: (p) => ({...p, type: 'simple', total: 50}), highlight: (p) => p.type === 'simple'},
    ])}
    asideWrapper={Playground.renderCheckbox([
        {label: '分页条数', onClick: (p) => ({...p, showPageSize: !p.showPageSize}), isChecked: (p) => p.showPageSize, isDisabled: (p) => p.type === 'simple'},
        {label: '快速跳转', onClick: (p) => ({...p, showJump: !p.showJump}), isChecked: (p) => p.showJump, isDisabled: (p) => p.type === 'simple'},
        {label: '总条数', onClick: (p) => ({...p, showTotal: !p.showTotal}), isChecked: (p) => p.showTotal, isDisabled: (p) => p.type === 'simple'},
    ])}
>
    <Pagination data-mark/>
</Playground>

## 基本分页
页数≤5 时
<Playground>
  <Pagination total={50}/>
</Playground>

## 更多分页
页数>5 时
<Playground>
  <Pagination total={100}/>
</Playground>

## 分页条数
改变每页显示的条目数。
<Playground>
  <Pagination
    total={250}
    showPageSize
  />
</Playground>

## 快速跳转页面
快速跳转到某一页。
<Playground>
  <Pagination
    total={250}
    showPageSize
    showTotal
    showJump/>
</Playground>

## 总条数
展示总条数。
<Playground>
  <Pagination
    total={250}
    showPageSize
    showTotal/>
</Playground>

## 简洁版本
简单的翻页，页数≤15时。
<Playground>
  <Pagination total={100} defaultCurrent={1} defaultPageSize={10} type='simple'/>
</Playground>
