## 受控组件
<Playground>
  {
    () => {
      const [page, setPage] = React.useState(1);
      const [pageSize, setPageSize] = React.useState(5);
      const total = 200;
      const changePage = (val) => {
        setPage(val)
      }
      return (
        <Pagination
          pageSize={pageSize}
          total={total}
          current={page}
          pageSizeOptions={[5, 10, 15, 20, 25, 30]}
          showPageSize
          onPageSizeChange={setPageSize}
          onChange={changePage}
        />
      )
    }
  }
</Playground>
