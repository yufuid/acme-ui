## 受控

<Playground>
    {
        () => {
            const [checked, setChecked] = React.useState(false);
            const [loading, setLoading] = React.useState(false);
            const onChange = (check, e) => {
                setLoading(true)
                const timer = setTimeout(() => {
                    setChecked(check);
                    setLoading(false);
                    clearTimeout(timer);
                }, 1000)
            }
            return <Switch checked={checked} onChange={onChange} loading={loading} />
        }
    }
</Playground>
