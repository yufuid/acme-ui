import { Playground, Props } from "~controls";
import { Button } from "@acme-ui/core";

<Props of={Button} />

<Props of={Button.Group} />

## 主要按钮

(重点突出)：用于主行动点，一个操作区域只能有一个主按钮。

<Playground>
  <div
    style={{ width: "100%", display: "flex", justifyContent: "space-between" }}
  >
    <Button>主要按钮</Button>
    <Button disabled>Disabled</Button>
    <Button loading>Loading</Button>
  </div>
</Playground>

## 次要按钮

(中等强调)：用于没有主次之分的一组行动点。

<Playground>
  <div
    style={{
      width: "100%",
      display: "flex",
      justifyContent: "space-between",
    }}
  >
    <Button mode="outlined">次要按钮</Button>
    <Button mode="outlined" disabled>
      Disabled
    </Button>
    <Button mode="outlined" loading>
      Loading
    </Button>
  </div>
</Playground>

## 文字按钮

(低强调)：文本按钮通常用于次要的操作。

<Playground>
  <div
    style={{ width: "100%", display: "flex", justifyContent: "space-between" }}
  >
    <Button mode="minimal">文字按钮</Button>
    <Button mode="minimal" disabled>
      Disabled
    </Button>
    <Button mode="minimal" loading>
      Loading
    </Button>
  </div>
</Playground>

## 危险按钮

(红色警告)：删除/移动/修改权限等危险操作，一般需要二次确认。

<Playground>
  <div
    style={{ width: "100%", display: "flex", justifyContent: "space-between" }}
  >
    <Button color="danger">危险按钮</Button>
    <Button color="danger" disabled>
      Disabled
    </Button>
    <Button color="danger" loading>
      Loading
    </Button>
  </div>
</Playground>

## 幽灵按钮

将其他按钮的内容反色，背景变为透明，常用在有色背景上。

<Playground>
  <div
    style={{
      backgroundColor: "#333333",
      display: "flex",
      justifyContent: "space-between",
      padding: "32px 40px",
    }}
  >
    <Button mode="contained" ghost>
      幽灵按钮
    </Button>
    <Button mode="outlined" ghost>
      幽灵按钮
    </Button>
    <Button mode="minimal" ghost>
      幽灵按钮
    </Button>
  </div>
</Playground>

## 按钮组

常用于多项类似操作。

<Playground>
  <div style={{display: 'flex', justifyContent: "center"}}>
    <Button.Group style={{marginRight: '80px'}} mode="minimal">
      <Button mode="outlined" size="large">
        1
      </Button>
      <Button mode="outlined">
        2
      </Button>
    </Button.Group>
    <Button.Group>
      <Button mode="contained">
        1
      </Button>
      <Button mode="contained">
        2
      </Button>
      <Button mode="contained">
        3
      </Button>
    </Button.Group>
  </div>
</Playground>

## Block 按钮

将按钮宽度调整为其父宽度的选项。

<Playground>
  <div>
    <Button style={{ marginBottom: "16px" }} mode="contained" fullWidth>
      主要按钮
    </Button>
    <Button style={{ marginBottom: "16px" }} mode="outlined" fullWidth>
      次要按钮
    </Button>
    <Button style={{ marginBottom: "16px" }} color="danger" fullWidth>
      危险按钮
    </Button>
    <Button mode="minimal" fullWidth>
      文字按钮
    </Button>
  </div>
</Playground>

## 按钮尺寸

按钮有大、标准、小三种尺寸。

<Playground>
  <div
    style={{
      width: "100%",
      display: "flex",
      justifyContent: "space-between",
      alignItems: "center",
    }}
  >
    <Button size="large">大尺寸</Button>
    <Button>标准尺寸</Button>
    <Button size="small">小尺寸</Button>
  </div>
</Playground>
