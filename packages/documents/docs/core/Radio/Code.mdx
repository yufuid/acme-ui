import { Playground, Props } from '~controls';
import { Radio, FormLabel } from '@acme-ui/core';

## 基础用法

由于选项默认可见，不宜过多，若选项过多，建议使用 Select 选择器

<Playground>
  {() => {
    const [checkedValue, setCheckedValue] = React.useState(false);
    const onChange = (e) => {
      const checked = e.target.checked;
      setCheckedValue(checked);
    };
    return (
      <div>
        <Radio onChange={onChange} checked={checkedValue} label="单选" />
      </div>
    );
  }}
</Playground>
